                      # 时间格式转换

                      function getFormatDate() {
    let date = new Date();
    let year = date.getFullYear();
    let month = date.getMonth() + 1;
    let day = date.getDate();
    let hour = date.getHours();
    let minutes = date.getMinutes();
    let seconds = date.getSeconds();
    month = (month<10)? '0'+ month : month;
    day = (day<10)? '0'+ day : day;
    hour = (hour<10)? '0'+ hour : hour;
    minutes = (minutes<10)? '0'+ minutes : minutes;
    seconds = (seconds<10)? '0'+ seconds : seconds;
    let currentDate = year + "-" + month + "-" + day
            + " " + hour + ":" + minutes + ":" + seconds;
    return currentDate;
}

                      # 自定义按钮

                      通过自定义按钮的方式，控制折线图/柱状图等对应数据的显示。
                      按钮的作用与echarts 自带的legend一致，只是样式不同。

        <li class="label-list pointer" v-for="(item,idx) in labelBox" :key="idx" @click="chooseLabel(item,idx)">
        <div class="label-shape"></div>
        <p class="label-btn" @click="toggleData(item,idx)">{{ item.name }}</p>
        </li>
    </ul>

                          // 对应数据组成的按钮组
    labelBox:[
        {checked:true,name:'前一天'},
        {checked:true,name:'今天'},
    ]
    chooseLabel(item,idx){
    setTimeout(() => {
        this.labelBox.forEach((v) => {
            this.myChart.dispatchAction({
                type: v.checked ? 'legendSelect' : 'legendUnSelect',
                // 图例名称
                name: v.name
            })
        })
    }, 0)

    }

                          dataURLtoBlob(dataurl) {
        var bstr = atob(dataurl),
            n = bstr.length,
            u8arr = new Uint8Array(n);
        while (n--) {
            u8arr[n] = bstr.charCodeAt(n);
        }
        return new Blob([u8arr], { type: 'audio/wav' });
    }
    this.playData='';
    this.playUrl = window.URL.createObjectURL(this.dataURLtoBlob(this.playData));

download(data){
    let reportTitle = '生产团队执行情况数据报表'
    const blob = new Blob([data], { type: 'application/vnd.ms-excel' })
    const href = URL.createObjectURL(blob) // 创建新的URL表示指定的blob对象
    const a = document.createElement('a') // 创建a标签
    a.style.display = 'none'
    a.href = href // 指定下载链接
    //设置指定元素上的某个属性值。如果属性已经存在，则更新该值
    a.setAttribute('download', reportTitle+'.xlsx')
    a.click() // 触发下载
    URL.revokeObjectURL(a.href) // 释放URL对象
    this.$message({
        type: 'success',
        message: '导出成功!'
    });
}

                      # 数组计算

                      计算多个数组的值对应相加后，得到的新数组

                      let arr1 = [1,2,3,4,5];
let arr2 = [11,22,33,44,55];
let arr3 = [111,222,333,444,555];
let sum = {arr1,arr2,arr3};
this.totalResult(sum);
totalResult(sum){
    //保存结果的数组
    var result = []; 
    for(var key in sum ){
        // 遍历 数组的每一项
        sum[key].map((val,idx)=>{
            if( result[idx] == null ||  result[idx] == ""){
                result[idx] = 0 ;
            }
            result[idx] += val ;
        })
    }
    console.log('计算的值',result)
    return result;
}

getNum(min, max) {
    min = Math.ceil(min);
    max = Math.floor(max);
    let result = Math.floor(Math.random() * (max - min + 1)) + min;
    return  result//含最大值，含最小值 
}

                      # chrome浏览器报错处理

                      报错内容：Added non-passive event listener to a scroll-blocking 'wheel' event. Consider marking event handler as 'passive' to make the page more responsive

                      将这一段js添加到main.js中，在vue挂在前执行即可。

                      //去除谷歌的touch事件警告
(function () {
  if (typeof EventTarget !== "undefined") {
    let func = EventTarget.prototype.addEventListener;
    EventTarget.prototype.addEventListener = function (type, fn, capture) {
      this.func = func;
      if (typeof capture !== "boolean") {
        capture = capture || {};
        capture.passive = false;
      }
      this.func(type, fn, capture);
    };
  };
}());
